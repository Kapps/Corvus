# A basic component, just to show-case that we can make a Sprite.
BlueSprite
	SpriteComponent (Sprite)
		Color: Color(White) # Note that this can use RGBA, RGB, NamedColor + A, or just NamedColor.

# Inherits BlueSprite, simply adding the Sprite itself.
PlayerSprite : BlueSprite
	SpriteComponent (Sprite)
		Sprite: Sprite(Sprites/TestPlayer)

Attributes
	AttributesComponent
		MaxHealth: Random(450, 500)
		Strength: 25
		Dexterity: 20

Healthy
	HealthBarComponent

Mover
	MovementComponent

Damager
	DamageComponent

StatusEffecter
	StatusEffectsComponent

Fighter
	CombatComponent
		AttackableEntities: Enum(CorvEngine.Components.EntityClassification, Enemy)

# Show off multiple inheritance / composition.
TestEntity : PlayerSprite, Healthy, Mover, Attributes, Damager, StatusEffecter, Fighter
	PhysicsComponent
	EquipmentComponent
		DefaultWeaponName: Bow
	ClassificationComponent
		Classification: Enum(CorvEngine.Components.EntityClassification, Player)
