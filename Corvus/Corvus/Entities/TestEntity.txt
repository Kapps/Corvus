# A basic component, just to show-case that we can make a Sprite.
BlueSprite
	SpriteComponent (Sprite)
		Color: Color(White) # Note that this can use RGBA, RGB, NamedColor + A, or just NamedColor.

# Inherits BlueSprite, simply adding the Sprite itself.
PlayerSprite : BlueSprite
	SpriteComponent (Sprite)
		Sprite: Sprite(Sprites/TestPlayer)

Healthy
	HealthComponent
		MaxHealth: Random(450, 500)

Mover
	MovementComponent

# Show off multiple inheritance / composition.
TestEntity : PlayerSprite, Healthy, Mover
